9 /**
10  * PIC = Personal Identity Code, "henkilotunnus" in Finnish.
11  *
12  * See https://vrk.fi/en/personal-identity-code1 for specs.
13  *
14  * The constructor is private on purpose, so that Pics can only
15  * be created via the companion object (the so called
16  * "smart constructor" pattern). This prevents instances which
17  * are in an illegal state. Every Pic object can thus be
18  * treated as a valid Pic.
19  *
20  * See the companion object for how to create instances of Pic.
21  *
22  * Some examples of [[Pic]] creation:
23  * {{{
24  * >>> Pic.fromString("290877-1639")
25  * Right(290877-1639)
26  *
27  * >>> Pic.fromString("290877-163")
28  * Left(Invalid PIC: '290877-163'. PIC should have 11 characters, but was 10 characters.)
29  *
30  * >>> Pic.fromString("290877-1638")
31  * Left(Invalid PIC: '290877-1638'. The control character ('8') is wrong: it should be '9'.)
32  *
33  * >>> Pic.fromStringUnsafe("290877-1639")
34  * 290877-1639
35  *
36  * // Pic.fromStringU is just a shorter alias for Pic.fromStringUnsafe.
37  * >>> Pic.fromStringU("290877-1639")
38  * 290877-1639
39  *
40  * // If you give Pic.fromStringUnsafe (or Pic.fromStringU) an invalid PIC, they throw an IllegalArgumentException.
41  * }}}
42  */

239   /**
240    * Create a Pic from an input String, returning an [[scala.util.Either]].
241    *
242    * See also [[fromStringUnsafe]] for cases where you are sure that the input is a valid PIC
243    * (or you are willing to handle exceptions).
244    *
245    * Examples:
246    * {{{
247    * >>> Pic.fromString("290877-1639")
248    * Right(290877-1639)
249    *
250    * >>> Pic.fromString("290877-163")
251    * Left(Invalid PIC: '290877-163'. PIC should have 11 characters, but was 10 characters.)
252    *
253    * >>> Pic.fromString("290877-1638")
254    * Left(Invalid PIC: '290877-1638'. The control character ('8') is wrong: it should be '9'.)
255    * }}}
256    *
257    * @param input a valid Personal Identity Code as a String.
258    * @return Left(String) if the given String is not a valid PIC, return a scala.util.Left containing an error message.
259    *         Right(Pic) if the given String is a valid PIC, return a scala.util.Right containing the [[Pic]] object.
260    */
261   def fromString(input: String): Either[String, Pic] = {
262     input.trim().toUpperCase match {
263       case s: String if s.length != 11 => Left(s"Invalid PIC: '${s}'. PIC should have 11 characters, but was ${s.length} characters.")
264       case s: String if s.length == 11 => createFromStringOfCorrectLength(input, s)
265     }
266   }

268   /**
269    * Create a Pic from an input String, throwing an [[scala.IllegalArgumentException]] if the input is not a valid PIC.
270    *
271    * See also:
272    *   - [[fromString]] for cases where you are not sure if the input is a valid PIC (for example,
273    * the input comes from a user), which returns an Either for you to handle.
274    *   - [[fromStringU]] for a shorter named alias of this function.
275    *
276    * Examples:
277    * {{{
278    * >>> Pic.fromStringUnsafe("290877-1639")
279    * 290877-1639
280    *
281    * >>> // Pic.fromStringUnsafe("foo") would throw an IllegalArgumentException.
282    * }}}
283    *
284    * @param input a valid Personal Identity Code as a String.
285    * @return a Pic object if the input is a valid PIC.
286    * @throws scala.IllegalArgumentException if the input is not a valid PIC.
287    */
288   def fromStringUnsafe(input: String): Pic = {
289     fromString(input) match {
290       case Left(errorMessage) => throw new IllegalArgumentException(errorMessage)
291       case Right(pic) => pic
292     }
293   }

309   /**
310    * Here we have certainty that parameter `cleanedInput` is 11 chars long.
311    */
312   private def createFromStringOfCorrectLength(originalInput: String, cleanedInput: String): Either[String, Pic] = {
313     // Note: These substring splits cannot fail, since the string is already confirmed as being 11 chars long.
314     val birthDatePart = cleanedInput.substring(0, 6)
315     val sign = cleanedInput.substring(6, 7)
316     val individualNumber = cleanedInput.substring(7, 10)
317     val controlCharacter = cleanedInput.substring(10, 11)
318     createFromSubstrings(PicParts(originalInput, cleanedInput, birthDatePart, sign, individualNumber, controlCharacter))
319   }

350   private val validationRules = List(birthDatePartMustBeNumeric, signMustHaveAcceptableValue, individualNumberMustBeNumeric)

352   /**
353    * Here we have certainty that all the substrings, when concatenated together, form a string of 11 chars.
354    */
355   private def createFromSubstrings(pp: PicParts): Either[String, Pic] = {
356     val failedValidationRule = validationRules.find(vr => !vr.predicate(pp))
357     failedValidationRule match {
358       case Some(rule) => Left(rule.errorMessageGenerator(pp))
359       case None => createFromValidParts(pp)
360     }
361   }
362 
363   /**
364    * Here we have certainty that all the parts are themselves valid, except for the control character.
365    */
366   private def createFromValidParts(pp: PicParts): Either[String, Pic] = {
367     // At this point the birthDatePart and individualNumber have been validated to be numeric, so we can just use .toLong directly without Try.
368     val expectedControlCharacter: Char = calculateExpectedControlCharacter((pp.birthDatePart + pp.individualNumber).toLong)
369     if (pp.controlCharacter.charAt(0) != expectedControlCharacter) {
370       Left(s"Invalid PIC: '${pp.originalInput}'. The control character ('${pp.controlCharacter}') is wrong: it should be '${expectedControlCharacter}'.")
371     } else {
372       val gender = if (pp.individualNumber.toInt % 2 == 0) Female else Male
373       val century = pp.sign.charAt(0) match {
374         case '+' => 1800
375         case '-' => 1900
376         case 'A' => 2000
377       }
378       val yearWithinCentury = pp.birthDatePart.toString.substring(4, 6).toInt
379       val birthYear = century + yearWithinCentury
380       val birthMonth = pp.birthDatePart.toString.substring(2, 4).toInt
381       val birthDay = pp.birthDatePart.toString.substring(0, 2).toInt
382       Right(new Pic(pp.cleanedInput, gender, birthYear, birthMonth, birthDay))
383     }
384   }

394    * concatenated with the individualNumber (163 in 290877-1639). In this example,

405   }
