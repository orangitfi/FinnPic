7 /**
8  * PIC = Personal Identity Code, "henkilotunnus" in Finnish.
9  *
10  * See https://vrk.fi/en/personal-identity-code1 for specs.
11  *
12  * The constructor is private on purpose, so that Pics can only
13  * be created via the companion object (the so called
14  * "smart constructor" pattern). This prevents instances which
15  * are in an illegal state. Every Pic object can thus be
16  * treated as a valid Pic.
17  *
18  * See the companion object for how to create instances of Pic.
19  *
20  * Some examples of [[Pic]] creation:
21  * {{{
22  * >>> Pic.fromString("290877-1639")
23  * Right(290877-1639)
24  *
25  * >>> Pic.fromString("290877-163")
26  * Left(Invalid PIC: '290877-163'. PIC should have 11 characters, but was 10 characters.)
27  *
28  * >>> Pic.fromString("290877-1638")
29  * Left(Invalid PIC: '290877-1638'. The control character ('8') is wrong: it should be '9'.)
30  *
31  * >>> Pic.fromStringUnsafe("290877-1639")
32  * 290877-1639
33  *
34  * // Pic.fromStringU is just a shorter alias for Pic.fromStringUnsafe.
35  * >>> Pic.fromStringU("290877-1639")
36  * 290877-1639
37  *
38  * // If you give Pic.fromStringUnsafe (or Pic.fromStringU) an invalid PIC, they throw an IllegalArgumentException.
39  * }}}
40  */

238   /**
239    * Create a Pic from an input String, returning an [[scala.util.Either]].
240    *
241    * See also [[fromStringUnsafe]] for cases where you are sure that the input is a valid PIC
242    * (or you are willing to handle exceptions).
243    *
244    * Examples:
245    * {{{
246    * >>> Pic.fromString("290877-1639")
247    * Right(290877-1639)
248    *
249    * >>> Pic.fromString("290877-163")
250    * Left(Invalid PIC: '290877-163'. PIC should have 11 characters, but was 10 characters.)
251    *
252    * >>> Pic.fromString("290877-1638")
253    * Left(Invalid PIC: '290877-1638'. The control character ('8') is wrong: it should be '9'.)
254    * }}}
255    *
256    * @param input a valid Personal Identity Code as a String.
257    * @return Left(String) if the given String is not a valid PIC, return a scala.util.Left containing an error message.
258    *         Right(Pic) if the given String is a valid PIC, return a scala.util.Right containing the [[Pic]] object.
259    */
260   def fromString(input: String): Either[String, Pic] = {
261     input.trim().toUpperCase match {
262       case s: String if s.length != 11 => Left(s"Invalid PIC: '${s}'. PIC should have 11 characters, but was ${s.length} characters.")
263       case s: String if s.length == 11 => createFromStringOfCorrectLength(input, s)
264     }
265   }
266 
267   /**
268    * Create a Pic from an input String, throwing an [[scala.IllegalArgumentException]] if the input is not a valid PIC.
269    *
270    * See also:
271    *   - [[fromString]] for cases where you are not sure if the input is a valid PIC (for example,
272    * the input comes from a user), which returns an Either for you to handle.
273    *   - [[fromStringU]] for a shorter named alias of this function.
274    *
275    * Examples:
276    * {{{
277    * >>> Pic.fromStringUnsafe("290877-1639")
278    * 290877-1639
279    *
280    * >>> // Pic.fromStringUnsafe("foo") would throw an IllegalArgumentException.
281    * }}}
282    *
283    * @param input a valid Personal Identity Code as a String.
284    * @return a Pic object if the input is a valid PIC.
285    * @throws scala.IllegalArgumentException if the input is not a valid PIC.
286    */
287   def fromStringUnsafe(input: String): Pic = {
288     fromString(input) match {
289       case Left(errorMessage) => throw new IllegalArgumentException(errorMessage)
290       case Right(pic) => pic
291     }
292   }

308   /**
309    * Here we have certainty that parameter `cleanedInput` is 11 chars long.
310    */
311   private def createFromStringOfCorrectLength(originalInput: String, cleanedInput: String): Either[String, Pic] = {
312     // Note: These substring splits cannot fail, since the string is already confirmed as being 11 chars long.
313     val birthDatePart = cleanedInput.substring(0, 6)
314     val sign = cleanedInput.substring(6, 7)
315     val individualNumber = cleanedInput.substring(7, 10)
316     val controlCharacter = cleanedInput.substring(10, 11)
317     createFromSubstrings(PicParts(originalInput, cleanedInput, birthDatePart, sign, individualNumber, controlCharacter))
318   }

320   private case class PicParts(originalInput: String,

322                               birthDatePart: String,
323                               sign: String,
324                               individualNumber: String,
325                               controlCharacter: String)

327   private case class ValidationRule(predicate: PicParts => Boolean, errorMessageGenerator: PicParts => String)

329   private val birthDatePartMustBeNumeric = ValidationRule(
330     pp =>
331       pp.birthDatePart.matches("\\d{6}"),
332     pp =>
333       s"Invalid PIC: '${pp.originalInput}'. The first six characters have to be numeric, but they were: '${pp.birthDatePart}'.")

335   private val signMustHaveAcceptableValue = ValidationRule(
336     pp =>
337       List('+', '-', 'A').contains(pp.sign.charAt(0)),

342   private val individualNumberMustBeNumeric = ValidationRule(
343     pp =>
344       Try(pp.individualNumber.toInt).toOption.isDefined,

349   private val validationRules = List(birthDatePartMustBeNumeric, signMustHaveAcceptableValue, individualNumberMustBeNumeric)
350 
351   /**
352    * Here we have certainty that all the substrings, when concatenated together, form a string of 11 chars.
353    */
354   private def createFromSubstrings(pp: PicParts): Either[String, Pic] = {
355     val failedValidationRule = validationRules.find(vr => !vr.predicate(pp))
356     failedValidationRule match {
357       case Some(rule) => Left(rule.errorMessageGenerator(pp))
358       case None => createFromValidParts(pp)
359     }
360   }
361 
362   /**
363    * Here we have certainty that all the parts are themselves valid, except for the control character.
364    */
365   private def createFromValidParts(pp: PicParts): Either[String, Pic] = {
366     // At this point the birthDatePart and individualNumber have been validated to be numeric, so we can just use .toLong directly without Try.
367     val expectedControlCharacter: Char = calculateExpectedControlCharacter((pp.birthDatePart + pp.individualNumber).toLong)
368     if (pp.controlCharacter.charAt(0) != expectedControlCharacter) {
369       Left(s"Invalid PIC: '${pp.originalInput}'. The control character ('${pp.controlCharacter}') is wrong: it should be '${expectedControlCharacter}'.")
370     } else {
371       val gender = if (pp.individualNumber.toInt % 2 == 0) Female else Male
372       val century = pp.sign.charAt(0) match {
373         case '+' => 1800
374         case '-' => 1900
375         case 'A' => 2000
376       }
377       val yearWithinCentury = pp.birthDatePart.toString.substring(4, 6).toInt
378       val birthYear = century + yearWithinCentury
379       val birthMonth = pp.birthDatePart.toString.substring(2, 4).toInt
380       val birthDay = pp.birthDatePart.toString.substring(0, 2).toInt
381       Right(new Pic(pp.cleanedInput, gender, birthYear, birthMonth, birthDay))
382     }
383   }

392    * The input to this method is the birth date part (290877 in 290877-1639)
393    * concatenated with the individualNumber (163 in 290877-1639). In this example,
394    * the returned char would be 9.
395    *
396    * See https://vrk.fi/en/personal-identity-code1 for more information about the control character calculation.

399    * @return the expected control character for the input given.
400    */

402     val remainder: Int = (input % 31).toInt
403     controlChars(remainder)
404   }
